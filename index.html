    <script>

        const rates = {

            colored: {

                headshot: { base: 300, add: 200, label: 'Bust-Up (หัว-อก)' },

                halfbody: { base: 400, add: 300, label: 'Half Body (ครึ่งตัว)' },

                fullbody: { base: 500, add: 400, label: 'Full Body (เต็มตัว)' }

            },

            lineart: {

                headshot: { base: 150, add: 100, label: 'Bust-Up (หัว-อก)' },

                halfbody: { base: 200, add: 150, label: 'Half Body (ครึ่งตัว)' },

                fullbody: { base: 300, add: 200, label: 'Full Body (เต็มตัว)' }

            }

        };


        let currentStyle = 'colored';

        let currentType = 'headshot';

        let extraChar = 0;


        function setStyle(style) {

            currentStyle = style;

            const btnColored = document.getElementById('btn-colored');

            const btnLineart = document.getElementById('btn-lineart');


            if(style === 'colored') {

                btnColored.className = "px-5 py-2 rounded-full transition-all bg-white shadow-sm font-medium";

                btnColored.style.color = "#7BAABD";

                btnLineart.className = "px-5 py-2 rounded-full transition-all text-gray-500";

                btnLineart.style.color = "";

            } else {

                btnLineart.className = "px-5 py-2 rounded-full transition-all bg-white shadow-sm font-medium";

                btnLineart.style.color = "#7BAABD";

                btnColored.className = "px-5 py-2 rounded-full transition-all text-gray-500";

                btnColored.style.color = "";

            }

            updateUI();

            calculate();

        }


        function selectType(type) {

            currentType = type;

            document.querySelectorAll('#base-cards > div').forEach(c => c.classList.remove('active-card'));

            document.getElementById(`card-${type}`).classList.add('active-card');

            updateUI();

            calculate();

        }


        function updateUI() {

            const r = rates[currentStyle];

            document.getElementById('price-headshot').innerText = r.headshot.base + ".-";

            document.getElementById('price-halfbody').innerText = r.halfbody.base + ".-";

            document.getElementById('price-fullbody').innerText = r.fullbody.base + ".-";

            document.getElementById('add-char-rate').innerText = `ราคาต่อตัว: ${r[currentType].add}.-`;

        }


        function changeChar(val) {

            extraChar = Math.max(0, extraChar + val);

            document.getElementById('char-count').innerText = extraChar;

            calculate();

        }


        function calculate() {

            const r = rates[currentStyle][currentType];

            let base = r.base;

            let charPrice = extraChar * r.add;

            let bgPrice = document.getElementById('opt-bg').checked ? 200 : 0;

            

            let total = base + charPrice + bgPrice;

            let commercialAdd = 0;


            let html = `

                <div class="flex justify-between">

                    <span>${currentStyle === 'colored' ? 'งานลงสี' : 'งานตัดเส้น'}: ${r.label}</span>

                    <span class="font-medium">${base.toLocaleString()}.-</span>

                </div>

            `;


            if(extraChar > 0) {

                html += `

                    <div class="flex justify-between text-gray-400">

                        <span>เพิ่มตัวละคร (${extraChar} ตัว)</span>

                        <span>+${charPrice.toLocaleString()}.-</span>

                    </div>

                `;

            }


            if(bgPrice > 0) {

                html += `

                    <div class="flex justify-between text-gray-400">

                        <span>เพิ่มฉากหลัง (เริ่มต้น)</span>

                        <span>+${bgPrice.toLocaleString()}.-</span>

                    </div>

                `;

            }


            if(document.getElementById('opt-comm').checked) {

                commercialAdd = total;

                total *= 2;

                html += `

                    <div class="flex justify-between font-medium pt-2" style="color: #7BAABD;">

                        <span>เรทเชิงพาณิชย์ (x2)</span>

                        <span>+${commercialAdd.toLocaleString()}.-</span>

                    </div>

                `;

            }


            document.getElementById('summary-list').innerHTML = html;

            document.getElementById('total-price').innerText = total.toLocaleString();

            document.getElementById('deposit-price').innerText = (total / 2).toLocaleString() + ".-";

        }


        function copyDetail() {

            const total = document.getElementById('total-price').innerText;

            const deposit = document.getElementById('deposit-price').innerText;

            const style = currentStyle === 'colored' ? 'งานลงสี' : 'งานตัดเส้น';

            const type = rates[currentStyle][currentType].label;

            

            let text = `🎨 รายละเอียดการจ้างคอมมิชชั่น\n`;

            text += `ประเภท: ${style} (${type})\n`;

            if(extraChar > 0) text += `เพิ่มตัวละคร: ${extraChar} ตัว\n`;

            if(document.getElementById('opt-bg').checked) text += `ออปชั่น: เพิ่มฉากหลัง (เริ่มต้น 200.-)\n`;

            text += `🏦 ยอดมัดจำ (50%): ${deposit}\n`;

            document.body.removeChild(el);
